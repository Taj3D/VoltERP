    54→import { fmtCurrency as _fmtCurrency, fmtNumber as _fmtNumber, fmtBDT as _fmtBDT } from "@/lib/number-format";
    55→
    56→const fmt = (v: any, type?: string) => {
    57→  if (v === null || v === undefined || v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v || "—";
    58→  if (type === "currency") return _fmtBDT(Number(v));
    59→  if (type === "date") { if (!v) return "—"; const dt = new Date(v); return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" }); }
    60→  if (type === "boolean") return v ? "Active" : "Inactive";
    61→  if (type === "percent") return `${Number(v).toFixed(2)}%`;
    62→  if (type === "number") return _fmtNumber(Number(v));
    63→  return String(v);
    64→};
    65→
    66→const fmtCurrency = (v: any) => {
    67→  if (v === null || v === undefined) return "—";
    68→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    69→  return _fmtBDT(Number(v));
    70→};

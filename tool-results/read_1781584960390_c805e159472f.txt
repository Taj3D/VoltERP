    50→
    51→const fmtCurrency = (v: any) => {
    52→  if (v === null || v === undefined) return "—";
    53→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    54→  return _fmtBDT(Number(v));
    55→};
    56→
    57→const fmtDate = (d: string | Date) => { if (!d) return "—"; const dt = new Date(d); return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" }); };
    58→
    59→const fmtPercent = (v: any) => {
    60→  if (v === null || v === undefined) return "—";
    61→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    62→  return `${Number(v).toFixed(2)}%`;
    63→};
    64→
    65→const safeNum = (v: any, fallback = 0) => {
    66→  const n = Number(v);
    67→  return isNaN(n) ? fallback : n;
    68→};
    69→
    70→// Safe number formatter — centralized utility guarantees Latin digits
    71→const fmtNum = (v: any) => _fmtNumberVal(safeNum(v));
    72→
    73→
    74→function vatMask(val: any, isVatAuditor: boolean) {
    75→  if (isVatAuditor) return "N/A (Audit Mode)";
    76→  return val;
    77→}

   210→export default function StockModulePage({ currentPage, isVatAuditor: propVat, userRole: propRole }: StockModulePageProps) {
   211→  const { toast } = useToast();
   212→  const isVatAuditor = propVat ?? false;
   213→  const userRole = propRole ?? "admin";
   214→  const isAdmin = userRole === "admin" || userRole === "manager";
   215→  const isSR = userRole === "sr";
   216→  const isDealer = userRole === "dealer";
   217→
   218→  // ─── Tab Mapping ───
   219→  const tabMap: Record<string, string> = {
   220→    "stock": "stock",
   221→    "stock-details": "stock-details",
   222→    "stock-transfers": "stock-transfers",
   223→    "opening-stock": "opening-stock",
   224→    "batch-master": "batch-master",
   225→    "valuation": "valuation",
   226→  };
   227→  const [activeTab, setActiveTab] = useState(tabMap[currentPage] || "stock");
   228→
   229→  useEffect(() => {
   230→    const mapped = tabMap[currentPage];
   231→    if (mapped && mapped !== activeTab) setActiveTab(mapped);
   232→  }, [currentPage]);
   233→
   234→  // ─── Shared Dropdowns ───
   235→  const [companies, setCompanies] = useState<any[]>([]);
   236→  const [products, setProducts] = useState<any[]>([]);
   237→  const [godowns, setGodowns] = useState<any[]>([]);
   238→  const [categories, setCategories] = useState<any[]>([]);
   239→  const [batches, setBatches] = useState<any[]>([]);
   240→
   241→  const loadDropdowns = useCallback(async () => {
   242→    try { const r = await apiFetch("/api/companies"); setCompanies(Array.isArray(r) ? r : []); } catch (e) { console.error("Failed to load companies:", e); }
   243→    try { const r = await apiFetch("/api/products"); setProducts(Array.isArray(r) ? r : []); } catch (e) { console.error("Failed to load products:", e); }
   244→    try { const r = await apiFetch("/api/godowns"); setGodowns(Array.isArray(r) ? r : []); } catch (e) { console.error("Failed to load godowns:", e); }
   245→    try { const r = await apiFetch("/api/categories"); setCategories(Array.isArray(r) ? r : []); } catch (e) { console.error("Failed to load categories:", e); }
   246→  }, []);
   247→
   248→  useEffect(() => { loadDropdowns(); }, [loadDropdowns]);
   249→
   250→  // ============================================================
   251→  // TAB 1: STOCK OVERVIEW STATE
   252→  // ============================================================
   253→
   254→  const [stockData, setStockData] = useState<any[]>([]);
   255→  const [stockLoading, setStockLoading] = useState(true);
   256→  const [stockSearch, setStockSearch] = useState("");
   257→  const [stockFilterGodown, setStockFilterGodown] = useState("all");
   258→  const [stockFilterCategory, setStockFilterCategory] = useState("all");
   259→  const [stockFilterStatus, setStockFilterStatus] = useState("all");
   260→  const [stockExpandedRows, setStockExpandedRows] = useState<Set<string>>(new Set());
   261→
   262→  const loadStock = useCallback(async () => {
   263→    setStockLoading(true);
   264→    try {
   265→      const params = new URLSearchParams();
   266→      if (stockFilterGodown && stockFilterGodown !== "all") params.set("godownId", stockFilterGodown);
   267→      if (stockFilterCategory && stockFilterCategory !== "all") params.set("categoryId", stockFilterCategory);
   268→      if (stockFilterStatus && stockFilterStatus !== "all") params.set("status", stockFilterStatus);
   269→      const qs = params.toString();
   270→      const res = await apiFetch(`/api/stock${qs ? `?${qs}` : ""}`);
   271→      setStockData(Array.isArray(res) ? res : []);
   272→    } catch (e: any) { toast({ title: "Error", description: e.message, variant: "destructive" }); }
   273→    finally { setStockLoading(false); }
   274→  }, [toast, stockFilterGodown, stockFilterCategory, stockFilterStatus]);
   275→
   276→  const stockStats = useMemo(() => ({
   277→    totalProducts: stockData.length,
   278→    inStock: stockData.filter(s => s.stockStatus === "In Stock").length,
   279→    lowStock: stockData.filter(s => s.stockStatus === "Low Stock").length,
   280→    outOfStock: stockData.filter(s => s.stockStatus === "Out of Stock").length,
   281→    totalValue: isVatAuditor ? "N/A (Audit Mode)" : stockData.reduce((s, i) => s + safeNum(i.stockValue), 0),
   282→    totalBatches: batches.length,
   283→  }), [stockData, batches, isVatAuditor]);
   284→
   285→  const stockFiltered = useMemo(() => {
   286→    let data = stockData;
   287→    if (stockSearch) {
   288→      const q = stockSearch.toLowerCase();
   289→      data = data.filter(s =>
   290→        (s.productCode || "").toLowerCase().includes(q) ||
   291→        (s.productName || "").toLowerCase().includes(q) ||
   292→        (s.category || "").toLowerCase().includes(q)
   293→      );
   294→    }
   295→    return data;
   296→  }, [stockData, stockSearch]);
   297→
   298→  const toggleStockExpand = useCallback((id: string) => {
   299→    setStockExpandedRows(prev => {
   300→      const next = new Set(prev);
   301→      if (next.has(id)) next.delete(id); else next.add(id);
   302→      return next;
   303→    });
   304→  }, []);
   305→
   306→  // ============================================================
   307→  // TAB 2: STOCK DETAILS STATE
   308→  // ============================================================
   309→
   310→  const [sdSelectedProduct, setSdSelectedProduct] = useState("");
   311→  const [sdData, setSdData] = useState<any[]>([]);
   312→  const [sdSummary, setSdSummary] = useState<any>(null);
   313→  const [sdProductInfo, setSdProductInfo] = useState<any>(null);
   314→  const [sdLoading, setSdLoading] = useState(false);
   315→  const [sdDateFrom, setSdDateFrom] = useState("");
   316→  const [sdDateTo, setSdDateTo] = useState("");
   317→
   318→  const loadStockDetails = useCallback(async (productId: string) => {
   319→    if (!productId) { setSdData([]); setSdSummary(null); setSdProductInfo(null); return; }
   320→    setSdLoading(true);
   321→    try {
   322→      const params = new URLSearchParams({ productId });
   323→      if (sdDateFrom) params.set("from", sdDateFrom);
   324→      if (sdDateTo) params.set("to", sdDateTo);
   325→      const res = await apiFetch(`/api/stock-details?${params.toString()}`);
   326→      if (res?.product) {
   327→        setSdProductInfo(res.product);
   328→        setSdData(res.entries || []);
   329→        setSdSummary(res.summary || null);
   330→      } else if (Array.isArray(res?.products)) {
   331→        // No product selected yet, just product list
   332→        setSdData([]);
   333→        setSdSummary(null);
   334→        setSdProductInfo(null);
   335→      } else {
   336→        setSdData(Array.isArray(res) ? res : []);
   337→        setSdSummary(null);
   338→        setSdProductInfo(null);
   339→      }
   340→    } catch (e: any) { toast({ title: "Error", description: e.message, variant: "destructive" }); }
   341→    finally { setSdLoading(false); }
   342→  }, [toast, sdDateFrom, sdDateTo]);
   343→
   344→  useEffect(() => {
   345→    if (activeTab === "stock-details" && sdSelectedProduct) {
   346→      loadStockDetails(sdSelectedProduct);
   347→    }
   348→  }, [activeTab, sdSelectedProduct, sdDateFrom, sdDateTo, loadStockDetails]);
   349→
   350→  // ============================================================
   351→  // TAB 3: STOCK TRANSFERS STATE
   352→  // ============================================================
   353→
   354→  const [trnData, setTrnData] = useState<any[]>([]);
   355→  const [trnLoading, setTrnLoading] = useState(true);
   356→  const [trnSearch, setTrnSearch] = useState("");
   357→  const [trnStatusFilter, setTrnStatusFilter] = useState("all");
   358→  const [trnShippingFilter, setTrnShippingFilter] = useState("all");
   359→  const [trnDialog, setTrnDialog] = useState(false);
   360→  const [trnEdit, setTrnEdit] = useState<any>(null);
   361→  const [trnSaving, setTrnSaving] = useState(false);
   362→  const [trnDelete, setTrnDelete] = useState<any>(null);
   363→  const [trnForm, setTrnForm] = useState<Record<string, any>>({
   364→    fromGodownId: "", toGodownId: "", date: new Date().toISOString().split("T")[0], status: "Pending", notes: "",

   366→  const [trnLines, setTrnLines] = useState<any[]>([{ productId: "", quantity: 1, batchId: "" }]);
   367→  const [trnExpandedRows, setTrnExpandedRows] = useState<Set<string>>(new Set());

   370→    setTrnLoading(true);
   371→    try {
   372→      const res = await apiFetch("/api/transfers");
   373→      setTrnData(Array.isArray(res) ? res : []);

   375→    finally { setTrnLoading(false); }
   376→  }, [toast]);
   377→
   378→  const trnStats = useMemo(() => {
   379→    const active = trnData.filter(t => t.isActive !== false);
   380→    return {
   381→      total: active.length,
   382→      pendingApproval: active.filter(t => t.status === "Pending").length,
   383→      inTransit: active.filter(t => t.shippingStatus === "In-Transit" || t.status === "In-Transit").length,
   384→      delivered: active.filter(t => t.shippingStatus === "Delivered" || t.status === "Delivered").length,
   385→      cancelled: active.filter(t => t.status === "Cancelled").length,
   386→    };
   387→  }, [trnData]);
   388→
   389→  const trnFiltered = useMemo(() => {
   390→    let data = trnData.filter(t => t.isActive !== false);
   391→    if (trnStatusFilter !== "all") data = data.filter(t => t.status === trnStatusFilter);
   392→    if (trnShippingFilter !== "all") data = data.filter(t => t.shippingStatus === trnShippingFilter);
   393→    if (trnSearch) {
   394→      const q = trnSearch.toLowerCase();
   395→      data = data.filter(t =>
   396→        (t.transferNo || "").toLowerCase().includes(q) ||
   397→        (t.fromGodown?.name || "").toLowerCase().includes(q) ||
   398→        (t.toGodown?.name || "").toLowerCase().includes(q)
   399→      );
   400→    }
   401→    return data;
   402→  }, [trnData, trnStatusFilter, trnShippingFilter, trnSearch]);
   403→
   404→  const toggleTrnExpand = useCallback((id: string) => {
   405→    setTrnExpandedRows(prev => {
   406→      const next = new Set(prev);
   407→      if (next.has(id)) next.delete(id); else next.add(id);
   408→      return next;
   409→    });
   410→  }, []);
   411→
   412→  const openTrnCreate = useCallback(() => {
   413→    setTrnEdit(null);
   414→    setTrnForm({
   415→      fromGodownId: "", toGodownId: "", date: new Date().toISOString().split("T")[0], status: "Pending", notes: "",
   416→    });
   417→    setTrnLines([{ productId: "", quantity: 1, batchId: "" }]);
   418→    setTrnDialog(true);
   419→  }, []);
   420→
   421→  const openTrnEdit = useCallback((item: any) => {
   422→    setTrnEdit(item);
   423→    setTrnForm({
   424→      fromGodownId: item.fromGodownId || "",
   425→      toGodownId: item.toGodownId || "",
   426→      date: item.date ? new Date(item.date).toISOString().split("T")[0] : "",
   427→      status: item.status || "Pending",
   428→      notes: item.notes || "",
   429→    });
   430→    setTrnLines(item.lines?.length > 0
   431→      ? item.lines.map((l: any) => ({ productId: l.productId, quantity: l.quantity, batchId: l.batchId || "", costPrice: l.costPrice || 0 }))
   432→      : [{ productId: "", quantity: 1, batchId: "" }]
   433→    );
   434→    setTrnDialog(true);
   435→  }, []);
   436→
   437→  const saveTrn = useCallback(async () => {
   438→    if (!trnForm.fromGodownId) { toast({ title: "Validation", description: "Source godown is required", variant: "destructive" }); return; }
   439→    if (!trnForm.toGodownId) { toast({ title: "Validation", description: "Destination godown is required", variant: "destructive" }); return; }
   440→    if (trnForm.fromGodownId === trnForm.toGodownId) { toast({ title: "Validation", description: "Source and destination cannot be the same", variant: "destructive" }); return; }

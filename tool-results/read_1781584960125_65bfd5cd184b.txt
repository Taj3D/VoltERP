    55→const fmtCurrency = (v: any) => {
    56→  if (v === null || v === undefined) return "—";
    57→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    58→  return _fmtBDT(Number(v));
    59→};
    60→
    61→const fmtDate = (d: string | Date) => { if (!d) return "—"; const dt = new Date(d); return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" }); };
    62→
    63→const fmtPercent = (v: any) => {
    64→  if (v === null || v === undefined) return "—";
    65→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    66→  return `${Number(v).toFixed(2)}%`;
    67→};
    68→
    69→const safeNum = (v: any, fallback = 0) => {
    70→  const n = Number(v);
    71→  return isNaN(n) ? fallback : n;
    72→};
    73→
    74→
    75→function vatMask(val: any, isVatAuditor: boolean) {
    76→  if (isVatAuditor) return "N/A (Audit Mode)";
    77→  return val;
    78→}

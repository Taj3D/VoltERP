    43→// Safe number formatter — avoids Bengali digits from toLocaleString
    44→const safeNumberFmt = new Intl.NumberFormat('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
    45→
    46→const fmt = (v: any, type?: string) => {
    47→  if (v === null || v === undefined || v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v || "—";
    48→  if (type === "currency") return `Tk. ${toLatinDigits(safeNumberFmt.format(Number(v)))}`;
    49→  if (type === "date") { if (!v) return "—"; const dt = new Date(v); return isNaN(dt.getTime()) ? "—" : toLatinDigits(dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" })); }
    50→  if (type === "boolean") return v ? "Active" : "Inactive";
    51→  return String(v);
    52→};
    53→
    54→const fmtDate = (d: string | Date) => { if (!d) return "—"; const dt = new Date(d); return isNaN(dt.getTime()) ? "—" : toLatinDigits(dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" })); };
    55→
    56→const fmtCurrency = (v: any) => {
    57→  if (v === null || v === undefined) return "—";
    58→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    59→  return `Tk. ${toLatinDigits(safeNumberFmt.format(Number(v)))}`;
    60→};
    61→
    62→const fmtPct = (v: any) => {
    63→  if (v === null || v === undefined) return "—";

   379→  const computeAmortization = useCallback((principal: number, annualRate: number, months: number) => {

   381→      // Simple linear schedule for zero-interest loans
   382→      const monthlyPayment = months > 0 ? principal / months : 0;
   383→      const schedule: { month: number; payment: number; principal: number; interest: number; balance: number }[] = [];
   384→      let balance = principal;
   385→      for (let i = 1; i <= months; i++) {
   386→        balance -= monthlyPayment;
   387→        schedule.push({ month: i, payment: monthlyPayment, principal: monthlyPayment, interest: 0, balance: Math.max(0, balance) });
   388→      }
   389→      return schedule;
   390→    }
   391→    const monthlyRate = annualRate / 100 / 12;
   392→    const emi = principal * monthlyRate * Math.pow(1 + monthlyRate, months) / (Math.pow(1 + monthlyRate, months) - 1);
   393→    const schedule: { month: number; payment: number; principal: number; interest: number; balance: number }[] = [];
   394→    let balance2 = principal;

   397→      const principalPart = emi - interest;
   398→      balance2 -= principalPart;
   399→      schedule.push({ month: i, payment: emi, principal: principalPart, interest, balance: Math.max(0, balance2) });
   400→    }

    40→// Safe number formatters — use Intl.NumberFormat to prevent Bengali digit output
    41→const currencyFormatter = new Intl.NumberFormat('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
    42→const numberFormatter = new Intl.NumberFormat('en-US');
    43→const dateFormatter = new Intl.DateTimeFormat('en-US', { day: '2-digit', month: 'short', year: 'numeric' });
    44→
    45→const fmt = (v: any, type?: string) => {
    46→  if (v === null || v === undefined || v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v || "—";
    47→  if (type === "currency") return `Tk. ${currencyFormatter.format(Number(v))}`;
    48→  if (type === "date") return v ? dateFormatter.format(new Date(v)) : "—";
    49→  if (type === "boolean") return v ? "Active" : "Inactive";
    50→  if (type === "number") return numberFormatter.format(Number(v));
    51→  return String(v);
    52→};
    53→
    54→const fmtCurrency = (v: any) => {
    55→  if (v === null || v === undefined) return "—";
    56→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    57→  return `Tk. ${currencyFormatter.format(Number(v))}`;
    58→};

    68→function getNestedValue(obj: any, path: string): any {
    69→  return path.split(".").reduce((acc, part) => acc?.[part], obj);

   560→        // Use LeaveAllocation data
   561→        const balances: Record<string, { used: number; entitled: number; remaining: number }> = {};

   566→            used: Number(alloc.usedDays) || 0,
   567→            entitled: Number(alloc.allocatedDays) || 0,

   569→          };

   616→      minAgeDate.setFullYear(minAgeDate.getFullYear() - 18);
   617→      if (dob > minAgeDate) {
   618→        const age = today.getFullYear() - dob.getFullYear();
   619→        setDobValidationError(`Personnel Validation Gate: Employee must be at least 18 years of age. Current age is approximately ${age} years. Minimum age requirement not met.`);
   620→      } else {
   621→        setDobValidationError("");
   622→      }
   623→    } else {
   624→      setDobValidationError("");
   625→    }
   626→  }, [config.key, formData.dateOfBirth]);
   627→
   628→  // Personnel Validation Gate: Watch leave form changes for balance deduction preview
   629→  useEffect(() => {
   630→    if (config.key === "employee-leaves" && formData.employeeId && formData.leaveType && formData.fromDate && formData.toDate) {
   631→      const from = new Date(formData.fromDate);
   632→      const to = new Date(formData.toDate);
   633→      if (to >= from) {
   634→        const requestedDays = Math.ceil((to.getTime() - from.getTime()) / (1000 * 60 * 60 * 24)) + 1;
   635→        const balance = leaveBalances[formData.leaveType];
   636→        if (balance) {
   637→          setBalanceDeductionPreview({
   638→            type: formData.leaveType,
   639→            currentRemaining: balance.remaining,
   640→            afterDeduction: balance.remaining - requestedDays,
   641→            isOver: balance.remaining < requestedDays,
   642→          });
   643→        } else {
   644→          setBalanceDeductionPreview(null);
   645→        }
   646→      } else {
   647→        setBalanceDeductionPreview(null);
   648→      }
   649→    } else {
   650→      setBalanceDeductionPreview(null);
   651→    }
   652→  }, [config.key, formData.employeeId, formData.leaveType, formData.fromDate, formData.toDate, leaveBalances]);
   653→
   654→  // Filtered data
   655→  const filteredData = useMemo(() => {
   656→    if (!search.trim()) return data;
   657→    const q = search.toLowerCase();
   658→    return data.filter(item =>
   659→      config.columns.some(col =>
   660→        String(getNestedValue(item, col.key) ?? "").toLowerCase().includes(q)
   661→      )
   662→    );
   663→  }, [data, search, config.columns]);
   664→
   665→  // KPI stats — enhanced with module-specific metrics
   666→  const stats = useMemo(() => {
   667→    const total = data.length;
   668→    const active = data.filter(d => d.isActive !== false).length;
   669→    const inactive = total - active;
   670→    // Employee-leave specific stats
   671→    const pendingLeaves = config.key === "employee-leaves" ? data.filter(d => d.status === "Pending").length : 0;
   672→    const approvedLeaves = config.key === "employee-leaves" ? data.filter(d => d.status === "Approved").length : 0;
   673→    const rejectedLeaves = config.key === "employee-leaves" ? data.filter(d => d.status === "Rejected").length : 0;
   674→    // Employee type breakdown
   675→    const permanentCount = config.key === "employees" ? data.filter(d => d.employeeType === "Permanent").length : 0;
   676→    const contractCount = config.key === "employees" ? data.filter(d => d.employeeType === "Contract").length : 0;
   677→    const probationCount = config.key === "employees" ? data.filter(d => d.employeeType === "Probation").length : 0;
   678→    // Total salary for employees (mask for VAT Auditor / SR)
   679→    const totalSalary = (config.key === "employees" && !isVatAuditor && !isSR)
   680→      ? data.filter(d => d.isActive !== false).reduce((sum, d) => sum + (Number(d.baseSalary) || 0), 0)
   681→      : 0;
   682→    // Credit status stats for customers/suppliers — use computed credit status
   683→    const creditFrozen = (config.key === "customers" || config.key === "suppliers") ? data.filter(d => (d.computedCreditStatus || d.creditStatus) === "Frozen").length : 0;
   684→    const creditOverLimit = (config.key === "customers" || config.key === "suppliers") ? data.filter(d => (d.computedCreditStatus || d.creditStatus) === "OverLimit").length : 0;
   685→    // AR/AP totals — use computed balance for accuracy
   686→    const totalOutstandingAR = (config.key === "customers" && !isVatAuditor)
   687→      ? data.filter(d => d.isActive !== false).reduce((sum, d) => {
   688→          const bal = d.computedCurrentBalance ?? Number(d.currentBalance) ?? 0;
   689→          const type = d.computedCurrentBalanceType ?? d.currentBalanceType;
   690→          return sum + (type === "Dr" ? bal : 0);
   691→        }, 0)
   692→      : 0;
   693→    const totalOutstandingAP = (config.key === "suppliers" && !isVatAuditor)
   694→      ? data.filter(d => d.isActive !== false).reduce((sum, d) => {
   695→          const bal = d.computedCurrentBalance ?? Number(d.currentBalance) ?? 0;
   696→          const type = d.computedCurrentBalanceType ?? d.currentBalanceType;
   697→          return sum + (type === "Cr" ? bal : 0);
   698→        }, 0)
   699→      : 0;
   700→    return { total, active, inactive, pendingLeaves, approvedLeaves, rejectedLeaves, permanentCount, contractCount, probationCount, totalSalary, creditFrozen, creditOverLimit, totalOutstandingAR, totalOutstandingAP };
   701→  }, [data, config.key]);
   702→
   703→  // Open create dialog
   704→  const openCreate = () => {

   706→    setCollisionError("");
   707→    const defaults: Record<string, any> = {};
   708→    config.formFields.forEach(f => {
   709→      if (f.defaultValue !== undefined) defaults[f.key] = f.defaultValue;
   710→      else if (f.type === "checkbox") defaults[f.key] = true;
   711→      else defaults[f.key] = "";

   714→    setDialogOpen(true);
   715→  };
   716→
   717→  // Open edit dialog
   718→  const openEdit = (item: any) => {

   720→    setCollisionError("");

   723→      const val = getNestedValue(item, f.key);
   724→      if (val !== undefined && val !== null) values[f.key] = val;
   725→      else if (f.defaultValue !== undefined) values[f.key] = f.defaultValue;
   726→      else values[f.key] = f.type === "checkbox" ? true : "";
   727→    });
   728→    setFormData(values);
   729→    setDialogOpen(true);
   730→  };
   731→
   732→  // Save (create or update) — Phase 8: Optimistic UI + snapshot rollback
   733→  const handleSave = async () => {
   734→    for (const f of config.formFields) {
   735→      if (f.required && !formData[f.key] && formData[f.key] !== 0) {
   736→        toast({ title: "Validation Error", description: `${f.label} is required`, variant: "destructive" });
   737→        return;
   738→      }
   739→    }
   740→
   741→    // Personnel Validation Gate: Standalone DOB 18+ age check
   742→    if (config.key === "employees" && formData.dateOfBirth) {
   743→      const dob = new Date(formData.dateOfBirth);
   744→      const today = new Date();
   745→      const minAgeDate = new Date(today);
   746→      minAgeDate.setFullYear(minAgeDate.getFullYear() - 18);
   747→      if (dob > minAgeDate) {
   748→        toast({ title: "Personnel Validation Gate", description: "Employee must be at least 18 years of age. Date of Birth indicates the individual is under the minimum age requirement.", variant: "destructive" });
   749→        return;
   750→      }
   751→    }
   752→
   753→    // Phase 8: Chronological Date Shield — Joining Date >= DOB + 18 years
   754→    if (config.key === "employees" && formData.dateOfBirth && formData.joiningDate) {
   755→      const dob = new Date(formData.dateOfBirth);
   756→      const joining = new Date(formData.joiningDate);
   757→      const minJoiningDate = new Date(dob);
   758→      minJoiningDate.setFullYear(minJoiningDate.getFullYear() + 18);
   759→      if (joining < minJoiningDate) {
   760→        toast({ title: "Date Validation Error", description: "Joining Date must be at least 18 years after Date of Birth", variant: "destructive" });
   761→        return;
   762→      }
   763→    }
   764→
   765→    // Phase 8: Chronological Date Shield — Resignation Date > Joining Date
   766→    if (config.key === "employees" && formData.resignationDate && formData.joiningDate) {
   767→      const joining = new Date(formData.joiningDate);
   768→      const resignation = new Date(formData.resignationDate);
   769→      if (resignation <= joining) {
   770→        toast({ title: "Date Validation Error", description: "Resignation/Termination Date must be after Joining Date", variant: "destructive" });
   771→        return;
   772→      }
   773→    }
   774→
   775→    // SR cannot modify credit limit on customers
   776→    if (shouldMaskCreditLimit && editingItem) {
   777→      delete formData.creditLimit;
   778→    }
   779→    // Case-insensitive duplicate name check before API call
   780→    const nameField = formData.name;
   781→    if (nameField && typeof nameField === 'string' && nameField.trim()) {
   782→      const normalizedName = nameField.trim().toLowerCase();
   783→      const duplicate = data.find(d => {
   784→        const existingName = (d.name || '').trim().toLowerCase();
   785→        return existingName === normalizedName && d.id !== editingItem?.id;
   786→      });
   787→      if (duplicate) {
   788→        toast({ title: "Corporate Entity Collision", description: `A ${singularize(config.label).toLowerCase()} with the name "${nameField.trim()}" already exists (case-insensitive match).`, variant: "destructive" });
   789→        return;
   790→      }
   791→    }
   792→    // Employee Leave: validate date range (toDate >= fromDate)
   793→    if (config.key === "employee-leaves" && formData.fromDate && formData.toDate) {
   794→      const from = new Date(formData.fromDate);
   795→      const to = new Date(formData.toDate);
   796→      if (to < from) {
   797→        toast({ title: "Validation Error", description: "End date cannot be before start date", variant: "destructive" });
   798→        return;
   799→      }
   800→      // Auto-calculate totalDays from date range
   801→      const diffMs = to.getTime() - from.getTime();
   802→      const calculatedDays = Math.max(1, Math.ceil(diffMs / (1000 * 60 * 60 * 24)) + 1);
   803→      formData.totalDays = calculatedDays;
   804→    }
   805→
   806→    // Phase 8: Snapshot for optimistic rollback
   807→    setPersonnelSnapshot([...data]);
   808→
   809→    // Phase 8: Optimistic mutation — immediately update local data (edits only; creates rely on loadData)
   810→    if (editingItem) {
   811→      setData(prev => prev.map(item => item.id === editingItem.id ? { ...item, ...formData } : item));
   812→    }
   813→
   814→    setSaving(true);
   815→    try {
   816→      if (editingItem) {
   817→        await apiFetch(`${config.apiPath}/${editingItem.id}`, {
   818→          method: "PUT",
   819→          body: JSON.stringify(formData),
   820→        });
   821→        toast({ title: "Updated", description: `${config.label} updated successfully` });
   822→      } else {
   823→        await apiFetch(config.apiPath, {
   824→          method: "POST",
   825→          body: JSON.stringify(formData),
   826→        });
   827→        toast({ title: "Created", description: `${config.label} created successfully` });
   828→      }
   829→      setDialogOpen(false);
   830→      loadData();
   831→    } catch (e: any) {
   832→      // Phase 8: Rollback on error
   833→      if (personnelSnapshot) {
   834→        setData(personnelSnapshot);
   835→        setPersonnelSnapshot(null);
   836→      } else {
   837→        loadData(); // Fallback: reload from server
   838→      }
   839→
   840→      // Phase 8: Corporate Entity Collision error handling
   841→      if (e.message && e.message.includes("Corporate Entity Collision")) {
   842→        setCollisionError(e.message);
   843→        setDialogOpen(true); // Re-open dialog to show collision banner
   844→      }
   845→
   846→      // Phase 8: Insufficient leave balance error
   847→      if (e.message && e.message.includes("Transaction Blocked: Insufficient accumulated leave balance")) {
   848→        toast({ title: "Leave Balance Insufficient", description: "Transaction Blocked: Insufficient accumulated leave balance.", variant: "destructive" });
   849→      } else {
   850→        toast({ title: "Error", description: e.message, variant: "destructive" });
   851→      }
   852→    } finally {
   853→      setSaving(false);
   854→    }
   855→  };
   856→
   857→  // Delete
   858→  const handleDelete = async (id: string) => {
   859→    try {
   860→      await apiFetch(`${config.apiPath}/${id}`, { method: "DELETE" });
   861→      toast({ title: "Deleted", description: `${config.label} deleted successfully` });
   862→      setDeleteConfirm(null);
   863→      loadData();
   864→    } catch (e: any) {
   865→      toast({ title: "Error", description: e.message, variant: "destructive" });
   866→    }
   867→  };
   868→
   869→  // Approve/Reject leave — Phase 8: Spin-lock + optimistic + balance refresh
   870→  const handleLeaveAction = async (id: string, action: "Approved" | "Rejected") => {
   871→    // Phase 8: Snapshot for rollback
   872→    setPersonnelSnapshot([...data]);
   873→
   874→    // Optimistic update
   875→    setData(prev => prev.map(item =>
   876→      item.id === id ? { ...item, status: action } : item
   877→    ));
   878→
   879→    try {
   880→      await apiFetch(`${config.apiPath}/${id}`, {
   881→        method: "PUT",
   882→        body: JSON.stringify({ status: action }),
   883→      });
   884→      toast({ title: `Leave ${action}`, description: `Leave request has been ${action.toLowerCase()}` });
   885→      loadData();
   886→      // Automated Leave Deductions: Refresh leave balance after approval/rejection
   887→      if (formData.employeeId) {
   888→        calculateLeaveBalances(formData.employeeId);
   889→      }
   890→    } catch (e: any) {
   891→      // Rollback
   892→      if (personnelSnapshot) {
   893→        setData(personnelSnapshot);
   894→        setPersonnelSnapshot(null);
   895→      } else {
   896→        loadData();
   897→      }
   898→      toast({ title: "Error", description: e.message, variant: "destructive" });
   899→    }
   900→  };
   901→
   902→  // Export CSV
   903→  const handleExportCSV = () => {
   904→    try {
   905→      const exportData = filteredData.map(item => {

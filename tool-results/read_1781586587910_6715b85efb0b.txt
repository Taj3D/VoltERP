    25→const safeFmt = new Intl.NumberFormat('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
    26→const fmt = (n: number) => safeFmt.format(n);
    27→
    28→const fmtDate = (d: string | Date) => {
    29→  if (!d) return "—";
    30→  const dt = new Date(d);
    31→  return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", {
    32→      day: "2-digit",
    33→      month: "short",
    34→      year: "numeric",
    35→      hour: "2-digit",
    36→      minute: "2-digit",
    37→    });
    38→};
    39→
    40→const fmtMs = (ms: number) => {
    41→  if (ms < 1000) return `${ms}ms`;
    42→  return `${(ms / 1000).toFixed(2)}s`;
    43→};

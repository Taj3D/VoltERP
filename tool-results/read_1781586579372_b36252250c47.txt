    38→const AUDIT_MASK = "N/A (Audit Mode)";
    39→
    40→const RESTRICTED_MASK = "N/A (Restricted)";
    41→
    42→import { fmtBDT as _fmtBDT } from "@/lib/number-format";
    43→
    44→const fmt = (v: any, type?: string) => {
    45→  if (String(v) === AUDIT_MASK || String(v) === RESTRICTED_MASK) return String(v);
    46→  if (v === null || v === undefined) return "—";
    47→  if (type === "currency") return _fmtBDT(Number(v));
    48→  if (type === "date") {
    49→    if (!v) return "—";
    50→    const dt = new Date(v);
    51→    return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" });
    52→  }
    53→  if (type === "percent") return `${Number(v).toFixed(2)}%`;
    54→  return String(v);
    55→};
    56→
    57→const fmtDate = (d: string | Date) => {
    58→  if (!d) return "—";
    59→  const dt = new Date(d);
    60→  return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" });
    61→};

  1725→function CoARow({
  1726→  account,
  1727→  level,
  1728→  expanded,
  1729→  onToggle,
  1730→  getChildren,
  1731→  isVatAuditor,
  1732→}: {
  1733→  account: any;
  1734→  level: number;
  1735→  expanded: Set<string>;
  1736→  onToggle: (id: string) => void;
  1737→  getChildren: (parentId: string) => any[];
  1738→  isVatAuditor: boolean;

  1740→  const children = getChildren(account.id);
  1741→  const hasChildren = children.length > 0;
  1742→  const isExpanded = expanded.has(account.id);
  1743→
  1744→  const classColor = (c: string) => {
  1745→    switch (c) {
  1746→      case "Asset": return "bg-blue-100 text-blue-700 dark:bg-blue-900/30 dark:text-blue-400";
  1747→      case "Liability": return "bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400";

  1752→    }
  1753→  };
  1754→
  1755→  return (
  1756→    <>
  1757→      <TableRow className={`hover:bg-muted/50 ${level === 0 ? "bg-slate-50/50 dark:bg-slate-800/20" : ""}`}>
  1758→        <TableCell className="w-8">
  1759→          {hasChildren ? (
  1760→            <button onClick={() => onToggle(account.id)} className="p-0.5 rounded hover:bg-muted">
  1761→              {isExpanded ? <ChevronDown className="w-3.5 h-3.5" /> : <ChevronRight className="w-3.5 h-3.5" />}

  1787→        <CoARow
  1788→          key={child.id}
  1789→          account={child}
  1790→          level={level + 1}
  1791→          expanded={expanded}
  1792→          onToggle={onToggle}

    48→const fmtCurrency = (v: any) => {
    49→  if (v === null || v === undefined) return "—";
    50→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    51→  return _fmtBDT(Number(v));
    52→};
    53→
    54→const fmtDate = (d: string | Date) => { if (!d) return "—"; const dt = new Date(d); return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" }); };
    55→
    56→const safeNum = (v: any, fallback = 0) => {
    57→  const n = Number(v);
    58→  return isNaN(n) ? fallback : n;
    59→};

    15→/** Helper to add cache headers to a response */
    16→function withCacheHeaders(data: unknown) {
    17→  return NextResponse.json(data, {
    18→    headers: {
    19→      'Cache-Control': 'private, max-age=10, stale-while-revalidate=30',
    20→    },
    21→  });
    22→}
    23→
    24→export async function GET(request: NextRequest) {
    25→  const security = await withApiSecurity(request, 'DashboardAnalytics', 'GET');
    26→  if (!security.authorized) return security.response;
    27→
    28→  try {
    29→    const { searchParams } = new URL(request.url);
    30→    const type = searchParams.get('type') || 'kpi';
    31→    const vatMode = searchParams.get('vatMode') === 'true';
    32→    const startDateStr = searchParams.get('startDate');
    33→    const endDateStr = searchParams.get('endDate');
    34→    const startDate = startDateStr ? new Date(startDateStr) : undefined;
    35→    const endDate = endDateStr ? new Date(endDateStr) : undefined;
    36→
    37→    // Multi-tenant isolation
    38→    const companyId = security.user.companyId;
    39→    const companyFilter = companyId ? { companyId } : {};
    40→
    41→    // Build date filter helper
    42→    const dateFilter = (baseWhere: Record<string, unknown>) => {
    43→      if (startDate || endDate) {
    44→        const dateConstraint: Record<string, unknown> = {};
    45→        if (startDate) dateConstraint.gte = startDate;
    46→        if (endDate) dateConstraint.lte = endDate;
    47→        baseWhere.date = dateConstraint;
    48→      }
    49→      return baseWhere;
    50→    };
    51→
    52→    switch (type) {
    53→      case 'kpi':
    54→        return await handleKPI(vatMode, dateFilter, companyFilter, security.user.role, security.user.id, security.user.name);
    55→      case 'monthly-trend':
    56→        return await handleMonthlyTrend(searchParams, vatMode, dateFilter, companyFilter, security.user.role, security.user.id, security.user.name);
    57→      case 'category-turnover':
    58→        return await handleCategoryTurnover(vatMode, dateFilter, companyFilter, security.user.role, security.user.id, security.user.name);
    59→      case 'stock-alerts':
    60→        return await handleStockAlerts(companyFilter);
    61→      case 'financial-ratios':
    62→        return await handleFinancialRatios(vatMode, dateFilter, companyFilter, security.user.role, security.user.id, security.user.name);
    63→      case 'top-performers':
    64→        return await handleTopPerformers(searchParams, vatMode, dateFilter, companyFilter, security.user.role, security.user.id, security.user.name);
    65→      case 'payment-mix':
    66→        return await handlePaymentMix(vatMode, dateFilter, companyFilter, security.user.role);
    67→      case 'daily-sales-trend':
    68→        return await handleDailySalesTrend(vatMode, dateFilter, companyFilter, security.user.role, security.user.id, security.user.name);
    69→      case 'receivables-aging':
    70→        return await handleReceivablesAging(vatMode, dateFilter, companyFilter, security.user.role, security.user.id, security.user.name);
    71→      default:
    72→        return NextResponse.json({ error: 'Invalid type parameter' }, { status: 400 });
    73→    }
    74→  } catch (error) {
    75→    console.error('Dashboard Analytics API error:', error);
    76→    return NextResponse.json({ error: 'Failed to generate analytics' }, { status: 500 });
    77→  }
    78→}

    80→// ─── KPI ─────────────────────────────────────────────────────────────────────
    81→async function handleKPI(
    82→  vatMode: boolean,
    83→  dateFilter: (base: Record<string, unknown>) => Record<string, unknown>,
    84→  companyFilter: Record<string, unknown>,
    85→  role: UserRole,
    86→  userId: string,
    87→  userName: string,
    88→) {
    89→  const now = new Date();
    90→  const todayStart = new Date(now.getFullYear(), now.getMonth(), now.getDate());
    91→  const monthStart = new Date(now.getFullYear(), now.getMonth(), 1);
    92→
    93→  // FIX 4: Exclude Cancelled in addition to Draft
    94→  const revenueStatusFilter = { notIn: ['Draft', 'Cancelled'] };
    95→
    96→  // Optimization: All queries in a SINGLE Promise.all (was 3 sequential rounds)
    97→  // Inlined calculateTotalReceivables/calculateTotalPayables to eliminate 10 redundant queries
    98→  // Also inlined YoY and sparkline queries that were in separate rounds
    99→  const lastYearStart = new Date(now.getFullYear() - 1, 0, 1);
   100→  const lastYearEnd = new Date(now.getFullYear() - 1, 11, 31);
   101→  const sevenDaysAgo = new Date(now.getFullYear(), now.getMonth(), now.getDate() - 6);
   102→
   103→  const [
   104→    salesAgg, purchasesAgg, expensesAgg, incomeAgg,
   105→    totalCustomers, totalSuppliers, totalProducts,
   106→    bankBalanceAgg,
   107→    todaysSalesAgg, todaysPurchasesAgg, todaysCollectionsAgg,
   108→    mtdSalesAgg, mtdPurchasesAgg,
   109→    allProductsForStock,
   110→    salesLinesForCOGS,
   111→    activeProducts,
   112→    // Inlined receivables data (was calculateTotalReceivables - 5 queries)
   113→    collectionsAgg, salesReturnsAgg, custOpenDrAgg, custOpenCrAgg,
   114→    // Inlined payables data (was calculateTotalPayables - 5 queries)
   115→    deliveriesAgg, purchaseReturnsAgg, suppOpenCrAgg, suppOpenDrAgg,
   116→    // YoY (was round 2)
   117→    lastYearSalesAgg, lastYearPurchasesAgg,
   118→    // Sparkline (was round 3)
   119→    sevenDaySales,
   120→  ] = await Promise.all([
   121→    // Core revenue aggregates
   122→    db.salesOrder.aggregate({
   123→      where: { ...dateFilter({ status: revenueStatusFilter, isActive: true, ...companyFilter }) },
   124→      _sum: { grandTotal: true },
   125→    }),
   126→    db.purchaseOrder.aggregate({
   127→      where: { ...dateFilter({ status: revenueStatusFilter, isActive: true, ...companyFilter }) },
   128→      _sum: { grandTotal: true },
   129→    }),
   130→    db.expense.aggregate({
   131→      where: { ...dateFilter({ isActive: true, ...companyFilter }) },
   132→      _sum: { amount: true },
   133→    }),
   134→    db.income.aggregate({
   135→      where: { ...dateFilter({ isActive: true, ...companyFilter }) },
   136→      _sum: { amount: true },
   137→    }),
   138→    // Counts
   139→    db.customer.count({ where: { isActive: true, ...companyFilter } }),
   140→    db.supplier.count({ where: { isActive: true, ...companyFilter } }),
   141→    db.product.count({ where: { isActive: true, ...companyFilter } }),
   142→    // Bank balance
   143→    db.bank.aggregate({
   144→      where: { ...companyFilter },
   145→      _sum: { currentBalance: true },
   146→    }),
   147→    // Today's aggregates
   148→    db.salesOrder.aggregate({
   149→      where: { date: { gte: todayStart }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   150→      _sum: { grandTotal: true },
   151→    }),
   152→    db.purchaseOrder.aggregate({
   153→      where: { date: { gte: todayStart }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   154→      _sum: { grandTotal: true },
   155→    }),
   156→    db.cashCollection.aggregate({
   157→      where: { date: { gte: todayStart }, isActive: true, ...companyFilter }, _sum: { amount: true },
   158→    }),
   159→    // Month-to-date aggregates
   160→    db.salesOrder.aggregate({
   161→      where: { date: { gte: monthStart }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   162→      _sum: { grandTotal: true },
   163→    }),
   164→    db.purchaseOrder.aggregate({
   165→      where: { date: { gte: monthStart }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   166→      _sum: { grandTotal: true },
   167→    }),
   168→    // Low stock products
   169→    db.product.findMany({
   170→      where: { isActive: true, ...companyFilter },
   171→      select: { openingStock: true, reorderLevel: true },
   172→    }),
   173→    // COGS: sales order lines with product cost price
   174→    db.salesOrderLine.findMany({
   175→      where: {
   176→        salesOrder: { status: { notIn: ['Draft', 'Cancelled'] }, isActive: true, ...companyFilter },
   177→      },
   178→      include: {
   179→        product: { select: { costPrice: true } },
   180→      },
   181→    }),
   182→    // Inventory value
   183→    db.product.findMany({
   184→      where: { isActive: true, ...companyFilter },
   185→      select: { costPrice: true, openingStock: true },
   186→    }),
   187→    // ── Inlined receivables data (eliminates 5 redundant queries) ──
   188→    db.cashCollection.aggregate({ where: { isActive: true, ...companyFilter }, _sum: { amount: true } }),
   189→    db.salesReturn.aggregate({ where: { isActive: true, ...companyFilter }, _sum: { grandTotal: true } }),
   190→    db.customer.aggregate({ where: { openingBalanceType: 'Dr', isActive: true, ...companyFilter }, _sum: { openingBalance: true } }),
   191→    db.customer.aggregate({ where: { openingBalanceType: 'Cr', isActive: true, ...companyFilter }, _sum: { openingBalance: true } }),
   192→    // ── Inlined payables data (eliminates 5 redundant queries) ──
   193→    db.cashDelivery.aggregate({ where: { isActive: true, ...companyFilter }, _sum: { amount: true } }),
   194→    db.purchaseReturn.aggregate({ where: { isActive: true, ...companyFilter }, _sum: { grandTotal: true } }),
   195→    db.supplier.aggregate({ where: { openingBalanceType: 'Cr', isActive: true, ...companyFilter }, _sum: { openingBalance: true } }),
   196→    db.supplier.aggregate({ where: { openingBalanceType: 'Dr', isActive: true, ...companyFilter }, _sum: { openingBalance: true } }),
   197→    // ── YoY data (was separate round 2) ──
   198→    db.salesOrder.aggregate({
   199→      where: { date: { gte: lastYearStart, lte: lastYearEnd }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   200→      _sum: { grandTotal: true },
   201→    }),
   202→    db.purchaseOrder.aggregate({
   203→      where: { date: { gte: lastYearStart, lte: lastYearEnd }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   204→      _sum: { grandTotal: true },
   205→    }),
   206→    // ── Sparkline data (was separate round 3) ──
   207→    db.salesOrder.findMany({
   208→      where: { date: { gte: sevenDaysAgo }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   209→      select: { date: true, grandTotal: true },
   210→    }),
   211→  ]);
   212→
   213→  // Compute derived values from parallel results
   214→  const lowStockProducts = allProductsForStock.filter(
   215→    (p) => Number(p.openingStock) <= Number(p.reorderLevel)
   216→  ).length;
   217→
   218→  const totalRevenue = Number(salesAgg._sum.grandTotal || 0);
   219→  const totalPurchases = Number(purchasesAgg._sum.grandTotal || 0);
   220→  const totalExpenses = Number(expensesAgg._sum.amount || 0);
   221→  const totalIncomes = Number(incomeAgg._sum.amount || 0);
   222→
   223→  const cogs = salesLinesForCOGS.reduce((sum, line) => {
   224→    const costPrice = Number(line.product?.costPrice || 0);
   225→    return safeFinancialAdd(sum, safeFinancialRound(Number(line.quantity) * costPrice));
   226→  }, 0);
   227→
   228→  const grossProfit = safeFinancialSubtract(totalRevenue, cogs);
   229→  const netProfit = safeFinancialSubtract(safeFinancialAdd(grossProfit, totalIncomes), totalExpenses);
   230→
   231→  const totalBankBalance = Number(bankBalanceAgg._sum.currentBalance || 0);
   232→
   233→  const totalInventoryValue = activeProducts.reduce(
   234→    (sum, p) => safeFinancialAdd(sum, safeFinancialRound(Number(p.costPrice) * Number(p.openingStock))),
   235→    0
   236→  );
   237→
   238→  // Compute receivables inline (was calculateTotalReceivables - eliminates re-querying salesAgg, etc.)
   239→  const receivablesBase = safeFinancialAdd(
   240→    Number(custOpenDrAgg._sum.openingBalance || 0),
   241→    safeFinancialSubtract(
   242→      totalRevenue,
   243→      safeFinancialAdd(
   244→        Number(collectionsAgg._sum.amount || 0),
   245→        safeFinancialAdd(
   246→          Number(salesReturnsAgg._sum.grandTotal || 0),
   247→          Number(custOpenCrAgg._sum.openingBalance || 0)
   248→        )
   249→      )
   250→    )
   251→  );
   252→  const totalReceivables = Math.max(0, receivablesBase);
   253→
   254→  // Compute payables inline (was calculateTotalPayables - eliminates re-querying purchasesAgg, etc.)
   255→  const payablesBase = safeFinancialAdd(
   256→    Number(suppOpenCrAgg._sum.openingBalance || 0),
   257→    safeFinancialSubtract(
   258→      totalPurchases,
   259→      safeFinancialAdd(
   260→        Number(deliveriesAgg._sum.amount || 0),
   261→        safeFinancialAdd(
   262→          Number(purchaseReturnsAgg._sum.grandTotal || 0),
   263→          Number(suppOpenDrAgg._sum.openingBalance || 0)
   264→        )
   265→      )
   266→    )
   267→  );
   268→  const totalPayables = Math.max(0, payablesBase);
   269→
   270→  const totalAssets = safeFinancialAdd(safeFinancialAdd(totalBankBalance, totalInventoryValue), totalReceivables);
   271→
   272→  const assetTurnoverRatio = totalAssets > 0 ? totalRevenue / totalAssets : 0;
   273→  const returnOnSales = totalRevenue > 0 ? (netProfit / totalRevenue) * 100 : 0;
   274→
   275→  const lastYearRevenue = Number(lastYearSalesAgg._sum.grandTotal || 0);
   276→  const lastYearPurchases = Number(lastYearPurchasesAgg._sum.grandTotal || 0);
   277→
   278→  // Sparkline
   279→  const sparklineData: number[] = [];
   280→  for (let i = 6; i >= 0; i--) {
   281→    const d = new Date(now.getFullYear(), now.getMonth(), now.getDate() - i);
   282→    const dayKey = d.toDateString();
   283→    const dayTotal = sevenDaySales
   284→      .filter(s => new Date(s.date).toDateString() === dayKey)
   285→      .reduce((sum, s) => sum + Number(s.grandTotal), 0);
   286→    sparklineData.push(dayTotal);
   287→  }
   288→
   289→  const result: Record<string, unknown> = {
   290→    totalRevenue,
   291→    totalPurchases,
   292→    totalExpenses,
   293→    totalIncomes,
   294→    grossProfit,
   295→    netProfit,
   296→    totalCustomers,
   297→    totalSuppliers,
   298→    totalProducts,
   299→    totalBankBalance,
   300→    totalReceivables,
   301→    totalPayables,
   302→    lowStockCount: lowStockProducts,
   303→    todaysSales: Number(todaysSalesAgg._sum.grandTotal || 0),
   304→    todaysPurchases: Number(todaysPurchasesAgg._sum.grandTotal || 0),
   305→    todaysCollections: Number(todaysCollectionsAgg._sum.amount || 0),
   306→    monthToDateSales: Number(mtdSalesAgg._sum.grandTotal || 0),
   307→    monthToDatePurchases: Number(mtdPurchasesAgg._sum.grandTotal || 0),
   308→    assetTurnoverRatio: safeFinancialRound(Math.round(assetTurnoverRatio * 100) / 100),
   309→    returnOnSales: safeFinancialRound(Math.round(returnOnSales * 100) / 100),
   310→    lastYearRevenue,
   311→    lastYearPurchases,
   312→    cashInHand: totalBankBalance,
   313→    inventoryValue: totalInventoryValue,
   314→    totalAssets,
   315→    sparklineData,
   316→  };
   317→
   318→  if (vatMode) {
   319→    result.vatMode = true;
   320→  }
   321→
   322→  // Stage 13: Apply deep recursive VAT Auditor masking
   323→  const maskedResult = maskDashboardForVatAuditor(result, role);
   324→
   325→  // Stage 13: Log user activity - fire-and-forget
   326→  logUserActivity({
   327→    action: 'EXPORT',
   328→    module: 'Audit-Dashboard-KPI',
   329→    userId,
   330→    userName,
   331→    details: 'Dashboard analytics report generated',
   332→  }).catch(console.error);
   333→
   334→  return withCacheHeaders(maskedResult);
   335→}

   337→// ─── Monthly Trend ───────────────────────────────────────────────────────────
   338→async function handleMonthlyTrend(
   339→  searchParams: URLSearchParams,
   340→  vatMode: boolean,
   341→  dateFilter: (base: Record<string, unknown>) => Record<string, unknown>,
   342→  companyFilter: Record<string, unknown>,
   343→  role: UserRole,
   344→  userId: string,
   345→  userName: string,
   346→) {
   347→  const months = parseInt(searchParams.get('months') || '12', 10);
   348→  const now = new Date();
   349→  const monthNames = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
   350→
   351→  const startDate = new Date(now.getFullYear(), now.getMonth() - months + 1, 1);
   352→
   353→  // FIX 4: Exclude Cancelled in addition to Draft
   354→  const revenueStatusFilter = { notIn: ['Draft', 'Cancelled'] };
   355→
   356→  // All with companyId filter
   357→  const [salesOrders, purchaseOrders, expenses, incomes] = await Promise.all([
   358→    db.salesOrder.findMany({
   359→      where: { date: { gte: startDate }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   360→      select: { date: true, grandTotal: true },
   361→    }),
   362→    db.purchaseOrder.findMany({
   363→      where: { date: { gte: startDate }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   364→      select: { date: true, grandTotal: true },
   365→    }),
   366→    db.expense.findMany({
   367→      where: { date: { gte: startDate }, isActive: true, ...companyFilter },
   368→      select: { date: true, amount: true },
   369→    }),
   370→    db.income.findMany({
   371→      where: { date: { gte: startDate }, isActive: true, ...companyFilter },
   372→      select: { date: true, amount: true },
   373→    }),
   374→  ]);
   375→
   376→  // Initialize monthly map
   377→  const monthlyMap: Record<string, { month: string; sales: number; purchases: number; expenses: number; income: number; net: number }> = {};
   378→  for (let i = months - 1; i >= 0; i--) {
   379→    const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
   380→    const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   381→    monthlyMap[key] = {
   382→      month: `${monthNames[d.getMonth()]} ${d.getFullYear()}`,
   383→      sales: 0,
   384→      purchases: 0,
   385→      expenses: 0,
   386→      income: 0,
   387→      net: 0,
   388→    };
   389→  }
   390→
   391→  // Aggregate sales
   392→  for (const so of salesOrders) {
   393→    const d = new Date(so.date);
   394→    const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   395→    if (monthlyMap[key]) monthlyMap[key].sales = safeFinancialAdd(monthlyMap[key].sales, Number(so.grandTotal));
   396→  }
   397→
   398→  // Aggregate purchases
   399→  for (const po of purchaseOrders) {
   400→    const d = new Date(po.date);
   401→    const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   402→    if (monthlyMap[key]) monthlyMap[key].purchases = safeFinancialAdd(monthlyMap[key].purchases, Number(po.grandTotal));
   403→  }
   404→
   405→  // Aggregate expenses
   406→  for (const exp of expenses) {
   407→    const d = new Date(exp.date);
   408→    const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   409→    if (monthlyMap[key]) monthlyMap[key].expenses = safeFinancialAdd(monthlyMap[key].expenses, Number(exp.amount));
   410→  }
   411→
   412→  // Aggregate incomes
   413→  for (const inc of incomes) {
   414→    const d = new Date(inc.date);
   415→    const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   416→    if (monthlyMap[key]) monthlyMap[key].income = safeFinancialAdd(monthlyMap[key].income, Number(inc.amount));
   417→  }
   418→
   419→  // Calculate net cash flow
   420→  for (const key of Object.keys(monthlyMap)) {
   421→    const m = monthlyMap[key];
   422→    m.net = safeFinancialSubtract(safeFinancialAdd(m.sales, m.income), safeFinancialAdd(m.purchases, m.expenses));
   423→  }
   424→
   425→  let data = Object.values(monthlyMap) as Record<string, unknown>[];
   426→
   427→  // Stage 13: Apply VAT Auditor masking for monthly trend
   428→  if (role === 'vat_auditor') {
   429→    data = data.map(d => maskDashboardForVatAuditor(d as Record<string, unknown>, role)) as Record<string, unknown>[];
   430→  }
   431→
   432→  // Stage 13: Log user activity - fire-and-forget
   433→  logUserActivity({
   434→    action: 'EXPORT',
   435→    module: 'Audit-Dashboard-KPI',
   436→    userId,
   437→    userName,
   438→    details: 'Dashboard analytics report generated',
   439→  }).catch(console.error);
   440→
   441→  return withCacheHeaders({ type: 'monthly-trend', data });
   442→}

   444→// ─── Category Turnover ───────────────────────────────────────────────────────
   445→async function handleCategoryTurnover(
   446→  vatMode: boolean,
   447→  dateFilter: (base: Record<string, unknown>) => Record<string, unknown>,
   448→  companyFilter: Record<string, unknown>,
   449→  role: UserRole,
   450→  userId: string,
   451→  userName: string,
   452→) {
   453→  // FIX 4: Exclude Cancelled in addition to Draft
   454→  const revenueStatusFilter = { notIn: ['Draft', 'Cancelled'] };
   455→
   456→  // Performance: Batch query all sales and purchase lines with product categories
   457→  // instead of N+1 per-category queries
   458→  const [categories, allSalesLines, allPurchaseLines] = await Promise.all([
   459→    db.category.findMany({
   460→      where: { isActive: true },
   461→      select: { id: true, name: true },
   462→    }),
   463→    db.salesOrderLine.findMany({
   464→      where: {
   465→        product: { isActive: true, ...companyFilter },
   466→        salesOrder: { status: revenueStatusFilter, isActive: true, ...companyFilter },
   467→      },
   468→      select: { quantity: true, total: true, product: { select: { categoryId: true } } },
   469→    }),
   470→    db.purchaseOrderLine.findMany({
   471→      where: {
   472→        product: { isActive: true, ...companyFilter },
   473→        purchaseOrder: { status: revenueStatusFilter, isActive: true, ...companyFilter },
   474→      },
   475→      select: { quantity: true, total: true, product: { select: { categoryId: true } } },
   476→    }),
   477→  ]);
   478→
   479→  // Aggregate by category in JavaScript (avoids N+1 queries)
   480→  const salesByCategory: Record<string, { qty: number; value: number }> = {};
   481→  const purchaseByCategory: Record<string, { qty: number; value: number }> = {};
   482→
   483→  for (const line of allSalesLines) {
   484→    const catId = line.product?.categoryId || 'uncategorized';
   485→    if (!salesByCategory[catId]) salesByCategory[catId] = { qty: 0, value: 0 };
   486→    salesByCategory[catId].qty += Number(line.quantity);
   487→    salesByCategory[catId].value = safeFinancialAdd(salesByCategory[catId].value, Number(line.total));
   488→  }
   489→
   490→  for (const line of allPurchaseLines) {
   491→    const catId = line.product?.categoryId || 'uncategorized';
   492→    if (!purchaseByCategory[catId]) purchaseByCategory[catId] = { qty: 0, value: 0 };
   493→    purchaseByCategory[catId].qty += Number(line.quantity);
   494→    purchaseByCategory[catId].value = safeFinancialAdd(purchaseByCategory[catId].value, Number(line.total));

   509→      value: sales.value,
   510→    };
   511→  });
   512→
   513→  // Sort by sales value descending, filter out zero
   514→  const filtered = data.filter((d) => typeof d.value === 'number' && d.value > 0);

   519→  if (role === 'vat_auditor') {
   520→    resultData = resultData.map(d => maskDashboardForVatAuditor(d, role));
   521→  }
   522→
   523→  // Stage 13: Log user activity - fire-and-forget
   524→  logUserActivity({
   525→    action: 'EXPORT',
   526→    module: 'Audit-Dashboard-KPI',
   527→    userId,
   528→    userName,
   529→    details: 'Dashboard analytics report generated',
   530→  }).catch(console.error);
   531→
   532→  return withCacheHeaders({ type: 'category-turnover', data: resultData });

   587→// ─── Financial Ratios ────────────────────────────────────────────────────────
   588→async function handleFinancialRatios(
   589→  vatMode: boolean,
   590→  dateFilter: (base: Record<string, unknown>) => Record<string, unknown>,
   591→  companyFilter: Record<string, unknown>,
   592→  role: UserRole,
   593→  userId: string,
   594→  userName: string,
   595→) {
   596→  // FIX 4: Exclude Cancelled in addition to Draft
   597→  const revenueStatusFilter = { notIn: ['Draft', 'Cancelled'] };
   598→
   599→  // Optimization: Single Promise.all with inlined receivables/payables data
   600→  // (was calling calculateTotalReceivables/calculateTotalPayables which re-queried data)
   601→  const [
   602→    salesAgg, purchasesAgg, expensesAgg, incomeAgg,
   603→    bankBalanceAgg, activeProducts, salesLinesForCOGS,
   604→    // Inlined receivables data (was calculateTotalReceivables - 5 queries)

   606→    // Inlined payables data (was calculateTotalPayables - 5 queries)
   607→    deliveriesAgg, purchaseReturnsAgg, suppOpenCrAgg, suppOpenDrAgg,
   608→  ] = await Promise.all([
   609→    db.salesOrder.aggregate({
   610→      where: { ...dateFilter({ status: revenueStatusFilter, isActive: true, ...companyFilter }) },
   611→      _sum: { grandTotal: true },
   612→    }),
   613→    db.purchaseOrder.aggregate({
   614→      where: { ...dateFilter({ status: revenueStatusFilter, isActive: true, ...companyFilter }) },
   615→      _sum: { grandTotal: true },
   616→    }),
   617→    db.expense.aggregate({
   618→      where: { ...dateFilter({ isActive: true, ...companyFilter }) },
   619→      _sum: { amount: true },
   620→    }),
   621→    db.income.aggregate({
   622→      where: { ...dateFilter({ isActive: true, ...companyFilter }) },
   623→      _sum: { amount: true },
   624→    }),
   625→    db.bank.aggregate({
   626→      where: { ...companyFilter },
   627→      _sum: { currentBalance: true },
   628→    }),
   629→    // Inventory value
   630→    db.product.findMany({
   631→      where: { isActive: true, ...companyFilter },
   632→      select: { costPrice: true, openingStock: true },
   633→    }),
   634→    // COGS
   635→    db.salesOrderLine.findMany({
   636→      where: {
   637→        salesOrder: { status: { notIn: ['Draft', 'Cancelled'] }, isActive: true, ...companyFilter },
   638→      },
   639→      include: {
   640→        product: { select: { costPrice: true } },
   641→      },
   642→    }),
   643→    // ── Inlined receivables data ──
   644→    db.cashCollection.aggregate({ where: { isActive: true, ...companyFilter }, _sum: { amount: true } }),
   645→    db.salesReturn.aggregate({ where: { isActive: true, ...companyFilter }, _sum: { grandTotal: true } }),
   646→    db.customer.aggregate({ where: { openingBalanceType: 'Dr', isActive: true, ...companyFilter }, _sum: { openingBalance: true } }),
   647→    db.customer.aggregate({ where: { openingBalanceType: 'Cr', isActive: true, ...companyFilter }, _sum: { openingBalance: true } }),
   648→    // ── Inlined payables data ──
   649→    db.cashDelivery.aggregate({ where: { isActive: true, ...companyFilter }, _sum: { amount: true } }),
   650→    db.purchaseReturn.aggregate({ where: { isActive: true, ...companyFilter }, _sum: { grandTotal: true } }),
   651→    db.supplier.aggregate({ where: { openingBalanceType: 'Cr', isActive: true, ...companyFilter }, _sum: { openingBalance: true } }),
   652→    db.supplier.aggregate({ where: { openingBalanceType: 'Dr', isActive: true, ...companyFilter }, _sum: { openingBalance: true } }),
   653→  ]);
   654→
   655→  const totalInventoryValue = activeProducts.reduce(
   656→    (sum, p) => safeFinancialAdd(sum, safeFinancialRound(Number(p.costPrice) * Number(p.openingStock))),
   657→    0
   658→  );
   659→
   660→  const totalRevenue = Number(salesAgg._sum.grandTotal || 0);
   661→  const totalPurchases = Number(purchasesAgg._sum.grandTotal || 0);
   662→  const totalExpenses = Number(expensesAgg._sum.amount || 0);
   663→  const totalIncomes = Number(incomeAgg._sum.amount || 0);
   664→
   665→  const cogs = salesLinesForCOGS.reduce((sum, line) => {
   666→    const costPrice = Number(line.product?.costPrice || 0);
   667→    return safeFinancialAdd(sum, safeFinancialRound(Number(line.quantity) * costPrice));
   668→  }, 0);
   669→
   670→  const grossProfit = safeFinancialSubtract(totalRevenue, cogs);
   671→  const netProfit = safeFinancialSubtract(safeFinancialAdd(grossProfit, totalIncomes), totalExpenses);
   672→
   673→  const totalBankBalance = Number(bankBalanceAgg._sum.currentBalance || 0);
   674→
   675→  // Compute receivables inline (eliminates calculateTotalReceivables re-query)
   676→  const receivablesBase = safeFinancialAdd(
   677→    Number(custOpenDrAgg._sum.openingBalance || 0),
   678→    safeFinancialSubtract(
   679→      totalRevenue,
   680→      safeFinancialAdd(
   681→        Number(collectionsAgg._sum.amount || 0),
   682→        safeFinancialAdd(
   683→          Number(salesReturnsAgg._sum.grandTotal || 0),
   684→          Number(custOpenCrAgg._sum.openingBalance || 0)
   685→        )
   686→      )
   687→    )
   688→  );
   689→  const totalReceivables = Math.max(0, receivablesBase);
   690→
   691→  // Compute payables inline (eliminates calculateTotalPayables re-query)
   692→  const payablesBase = safeFinancialAdd(
   693→    Number(suppOpenCrAgg._sum.openingBalance || 0),
   694→    safeFinancialSubtract(
   695→      totalPurchases,
   696→      safeFinancialAdd(
   697→        Number(deliveriesAgg._sum.amount || 0),
   698→        safeFinancialAdd(
   699→          Number(purchaseReturnsAgg._sum.grandTotal || 0),
   700→          Number(suppOpenDrAgg._sum.openingBalance || 0)
   701→        )
   702→      )
   703→    )
   704→  );
   705→  const totalPayables = Math.max(0, payablesBase);
   706→
   707→  // Current assets = bank + inventory + receivables
   708→  const currentAssets = safeFinancialAdd(safeFinancialAdd(totalBankBalance, totalInventoryValue), totalReceivables);
   709→  // Current liabilities = payables
   710→  const currentLiabilities = totalPayables;
   711→
   712→  // Total assets (simplified: current assets)
   713→  const totalAssets = currentAssets;
   714→
   715→  // Total liabilities (simplified: payables)
   716→  const totalLiab = totalPayables;
   717→  // Total equity = assets - liabilities
   718→  const totalEquity = safeFinancialSubtract(totalAssets, totalLiab);
   719→
   720→  // Ratios
   721→  const currentRatio = currentLiabilities > 0 ? currentAssets / currentLiabilities : 0;
   722→  const debtToEquityRatio = totalEquity > 0 ? totalLiab / totalEquity : 0;
   723→  const grossProfitMargin = totalRevenue > 0 ? (grossProfit / totalRevenue) * 100 : 0;
   724→  const netProfitMargin = totalRevenue > 0 ? (netProfit / totalRevenue) * 100 : 0;
   725→  const receivablesTurnover = totalReceivables > 0 ? totalRevenue / totalReceivables : 0;
   726→  const payablesTurnover = totalPayables > 0 ? totalPurchases / totalPayables : 0;
   727→  const inventoryTurnover = totalInventoryValue > 0 ? cogs / totalInventoryValue : 0;
   728→  const assetTurnover = totalAssets > 0 ? totalRevenue / totalAssets : 0;
   729→  const workingCapital = safeFinancialSubtract(currentAssets, currentLiabilities);
   730→  const quickRatio = currentLiabilities > 0 ? safeFinancialSubtract(currentAssets, totalInventoryValue) / currentLiabilities : 0;
   731→
   732→  const result: Record<string, unknown> = {
   733→    type: 'financial-ratios',
   734→    currentRatio: safeFinancialRound(Math.round(currentRatio * 100) / 100),
   735→    debtToEquityRatio: safeFinancialRound(Math.round(debtToEquityRatio * 100) / 100),
   736→    grossProfitMargin: safeFinancialRound(Math.round(grossProfitMargin * 100) / 100),
   737→    netProfitMargin: safeFinancialRound(Math.round(netProfitMargin * 100) / 100),
   738→    receivablesTurnover: safeFinancialRound(Math.round(receivablesTurnover * 100) / 100),
   739→    payablesTurnover: safeFinancialRound(Math.round(payablesTurnover * 100) / 100),
   740→    inventoryTurnover: safeFinancialRound(Math.round(inventoryTurnover * 100) / 100),
   741→    assetTurnover: safeFinancialRound(Math.round(assetTurnover * 100) / 100),
   742→    workingCapital: safeFinancialRound(Math.round(workingCapital * 100) / 100),
   743→    quickRatio: safeFinancialRound(Math.round(quickRatio * 100) / 100),
   744→  };
   745→
   746→  // Stage 13: Apply deep recursive VAT Auditor masking
   747→  const maskedResult = maskDashboardForVatAuditor(result, role);
   748→
   749→  // Stage 13: Log user activity - fire-and-forget
   750→  logUserActivity({
   751→    action: 'EXPORT',
   752→    module: 'Audit-Dashboard-KPI',
   753→    userId,
   754→    userName,
   755→    details: 'Dashboard analytics report generated',
   756→  }).catch(console.error);
   757→
   758→  return withCacheHeaders(maskedResult);
   759→}

   955→// ─── Daily Sales Trend (last 30 days) ────────────────────────────────────────
   956→async function handleDailySalesTrend(
   957→  vatMode: boolean,
   958→  dateFilter: (base: Record<string, unknown>) => Record<string, unknown>,
   959→  companyFilter: Record<string, unknown>,
   960→  role: UserRole,
   961→  userId: string,
   962→  userName: string,
   963→) {

   967→
   968→  const [salesOrders, expenses] = await Promise.all([
   969→    db.salesOrder.findMany({
   970→      where: { date: { gte: thirtyDaysAgo }, status: revenueStatusFilter, isActive: true, ...companyFilter },
   971→      select: { date: true, grandTotal: true },

   976→    }),
   977→  ]);
   978→
   979→  // Initialize daily map for last 30 days
   980→  const dailyMap: Record<string, { date: string; sales: number; expenses: number }> = {};
   981→  for (let i = 29; i >= 0; i--) {
   982→    const d = new Date(now.getFullYear(), now.getMonth(), now.getDate() - i);
   983→    const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;

   985→      date: new Intl.DateTimeFormat('en-GB', { day: '2-digit', month: 'short' }).format(d),
   986→      sales: 0,
   987→      expenses: 0,
   988→    };
   989→  }
   990→
   991→  for (const so of salesOrders) {
   992→    const d = new Date(so.date);
   993→    const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;
   994→    if (dailyMap[key]) dailyMap[key].sales = safeFinancialAdd(dailyMap[key].sales, Number(so.grandTotal));
   995→  }
   996→
   997→  for (const exp of expenses) {
   998→    const d = new Date(exp.date);
   999→    const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;
  1000→    if (dailyMap[key]) dailyMap[key].expenses = safeFinancialAdd(dailyMap[key].expenses, Number(exp.amount));
  1001→  }
  1002→
  1003→  let data = Object.values(dailyMap) as Record<string, unknown>[];
  1004→
  1005→  // Apply VAT Auditor masking
  1006→  if (role === 'vat_auditor') {
  1007→    data = data.map(d => maskDashboardForVatAuditor(d, role)) as Record<string, unknown>[];
  1008→  }
  1009→
  1010→  // Log user activity - fire-and-forget
  1011→  logUserActivity({
  1012→    action: 'EXPORT',
  1013→    module: 'Audit-Dashboard-KPI',
  1014→    userId,
  1015→    userName,
  1016→    details: 'Dashboard analytics report generated',
  1017→  }).catch(console.error);
  1018→
  1019→  return withCacheHeaders({ type: 'daily-sales-trend', data });
  1020→}

  1022→// ─── Receivables Aging ───────────────────────────────────────────────────────
  1023→async function handleReceivablesAging(
  1024→  vatMode: boolean,
  1025→  dateFilter: (base: Record<string, unknown>) => Record<string, unknown>,
  1026→  companyFilter: Record<string, unknown>,
  1027→  role: UserRole,
  1028→  userId: string,
  1029→  userName: string,
  1030→) {
  1031→  const today = new Date();
  1032→
  1033→  // FIX 4: Exclude Cancelled in addition to Draft
  1034→  const revenueStatusFilter = { notIn: ['Draft', 'Cancelled'] };
  1035→
  1036→  // Optimization: Run all queries in a single parallel batch
  1037→  // (previously salesOrders was sequential, then cashCollections/salesReturns in parallel,
  1038→  // then customerOpeningDr/Cr sequential - now all in one Promise.all)
  1039→  const [salesOrders, cashCollections, salesReturns, customerOpeningDr, customerOpeningCr] = await Promise.all([
  1040→    db.salesOrder.findMany({
  1041→      where: { status: revenueStatusFilter, isActive: true, ...companyFilter },
  1042→      select: {
  1043→        id: true,
  1044→        customerId: true,
  1045→        date: true,
  1046→        grandTotal: true,
  1047→      },
  1048→      orderBy: { date: 'asc' },
  1049→    }),
  1050→    db.cashCollection.findMany({
  1051→      where: { isActive: true, status: { not: 'Draft' }, ...companyFilter },
  1052→      select: { customerId: true, amount: true },
  1053→    }),
  1054→    db.salesReturn.findMany({
  1055→      where: { isActive: true, status: { not: 'Draft' }, ...companyFilter },
  1056→      select: { salesOrderId: true, grandTotal: true },
  1057→    }),
  1058→    db.customer.aggregate({
  1059→      where: { openingBalanceType: 'Dr', isActive: true, ...companyFilter },
  1060→      _sum: { openingBalance: true },
  1061→    }),
  1062→    db.customer.aggregate({
  1063→      where: { openingBalanceType: 'Cr', isActive: true, ...companyFilter },
  1064→      _sum: { openingBalance: true },
  1065→    }),
  1066→  ]);
  1067→
  1068→  // Build a map of collections per customer
  1069→  const collectionsByCustomer: Record<string, number> = {};
  1070→  for (const cc of cashCollections) {
  1071→    collectionsByCustomer[cc.customerId] = safeFinancialAdd(collectionsByCustomer[cc.customerId] || 0, Number(cc.amount));
  1072→  }
  1073→
  1074→  // Build a map of returns per SO
  1075→  const returnsBySO: Record<string, number> = {};
  1076→  for (const sr of salesReturns) {
  1077→    returnsBySO[sr.salesOrderId] = safeFinancialAdd(returnsBySO[sr.salesOrderId] || 0, Number(sr.grandTotal));
  1078→  }
  1079→
  1080→  let current = 0;
  1081→  let days31to60 = 0;
  1082→  let days61to90 = 0;
  1083→  let days90plus = 0;
  1084→
  1085→  // Track per-customer collection allocation
  1086→  const customerCollectionUsed: Record<string, number> = {};
  1087→
  1088→  for (const so of salesOrders) {
  1089→    const soTotal = Number(so.grandTotal);
  1090→    const returned = returnsBySO[so.id] || 0;
  1091→
  1092→    // Allocate collections to this SO (FIFO by customer)
  1093→    const custAvailable = collectionsByCustomer[so.customerId] || 0;
  1094→    const custUsed = customerCollectionUsed[so.customerId] || 0;
  1095→    const custRemaining = safeFinancialSubtract(custAvailable, custUsed);
  1096→
  1097→    const collectedForThis = Math.min(safeFinancialSubtract(soTotal, returned), custRemaining);
  1098→    customerCollectionUsed[so.customerId] = safeFinancialAdd(custUsed, Math.max(0, collectedForThis));
  1099→
  1100→    const balance = safeFinancialSubtract(soTotal, safeFinancialAdd(returned, Math.max(0, collectedForThis)));
  1101→
  1102→    if (balance > 0) {
  1103→      const daysOutstanding = Math.floor(
  1104→        (today.getTime() - new Date(so.date).getTime()) / (1000 * 60 * 60 * 24)
  1105→      );
  1106→
  1107→      if (daysOutstanding <= 30) {
  1108→        current = safeFinancialAdd(current, balance);
  1109→      } else if (daysOutstanding <= 60) {
  1110→        days31to60 = safeFinancialAdd(days31to60, balance);
  1111→      } else if (daysOutstanding <= 90) {
  1112→        days61to90 = safeFinancialAdd(days61to90, balance);
  1113→      } else {
  1114→        days90plus = safeFinancialAdd(days90plus, balance);
  1115→      }
  1116→    }
  1117→  }
  1118→
  1119→  // Use already-fetched opening balance data (was sequential queries before)
  1120→  const openingDrTotal = Number(customerOpeningDr._sum.openingBalance || 0);
  1121→  const openingCrTotal = Number(customerOpeningCr._sum.openingBalance || 0);
  1122→  const openingBalanceNet = safeFinancialSubtract(openingDrTotal, openingCrTotal);
  1123→
  1124→  // Opening balance receivables are typically aged (assume 90+ days as they're historical)
  1125→  if (openingBalanceNet > 0) {
  1126→    days90plus = safeFinancialAdd(days90plus, openingBalanceNet);
  1127→  }
  1128→
  1129→  const totalOutstanding = safeFinancialAdd(safeFinancialAdd(safeFinancialAdd(current, days31to60), days61to90), days90plus);
  1130→
  1131→  const result: Record<string, unknown> = {
  1132→    type: 'receivables-aging',
  1133→    totalOutstanding,
  1134→    current,
  1135→    days31to60,
  1136→    days61to90,
  1137→    days90plus,
  1138→    openingBalance: openingBalanceNet,
  1139→    vatMode: vatMode || undefined,
  1140→  };
  1141→
  1142→  // Stage 13: Apply deep recursive VAT Auditor masking
  1143→  const maskedResult = maskDashboardForVatAuditor(result, role);
  1144→
  1145→  // Stage 13: Log user activity - fire-and-forget
  1146→  logUserActivity({
  1147→    action: 'EXPORT',
  1148→    module: 'Audit-Dashboard-KPI',
  1149→    userId,
  1150→    userName,
  1151→    details: 'Dashboard analytics report generated',
  1152→  }).catch(console.error);
  1153→
  1154→  return withCacheHeaders(maskedResult);
  1155→}

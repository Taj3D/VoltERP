    41→import { fmtBDT as _fmtBDT, fmtNumber as _fmtNumberVal } from "@/lib/number-format";
    42→
    43→const fmtCurrency = (v: any) => {
    44→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    45→  const num = Number(v);
    46→  if (isNaN(num) || v === null || v === undefined) return "—";
    47→  return _fmtBDT(num);
    48→};
    49→
    50→const fmt = (v: any, type?: string) => {
    51→  if (v === null || v === undefined) return "—";
    52→  if (type === "currency") return fmtCurrency(v);
    53→  if (type === "date") { if (!v) return "—"; const dt = new Date(v); return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" }); }
    54→  if (type === "boolean") return v ? "Active" : "Inactive";
    55→  if (type === "number") {
    56→    const num = Number(v);
    57→    if (isNaN(num)) return "—";
    58→    return _fmtNumberVal(num);
    59→  }
    60→  return String(v);
    61→};
    62→
    63→const fmtEmpty = (v: any) => {
    64→  if (v === null || v === undefined || v === '') return '—';
    65→  return String(v);
    66→};
    67→
    68→const fmtDate = (d: string | Date) => { if (!d) return "—"; const dt = new Date(d); return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" }); };

    70→// SMS Character Bounds Computation (Client-side mirror)
    71→const computeClientSmsSegments = (message: string) => {
    72→  const charCount = message.length;
    73→  const isUnicode = /[^\x00-\x7F]/.test(message);
    74→  const charsPerSegment = isUnicode ? 70 : 160;
    75→  const segmentCount = charCount > 0 ? Math.ceil(charCount / charsPerSegment) : 1;
    76→  return { charCount, isUnicode, segmentCount, charsPerSegment };
    77→};

   514→  const handleSendSms = async () => {

   538→      const optRate = optIsUnicode ? unicodeCostPerSegment : costPerSegment;

   545→        charCount: optCharCount,

   550→        cost: estimatedCost,

   553→      };

   555→      // 2. Clear the form immediately (like WhatsApp clears the input)
   556→      const savedRecipient = smsRecipient.trim();
   557→      const savedMessage = smsMessage.trim();
   558→      setSmsRecipient("");
   559→      setSmsMessage("");
   560→
   561→      // 3. Add optimistic entry to the top of the logs
   562→      setSmsLogs((prev) => [optimisticLog, ...prev]);
   563→

   565→      try {
   566→        const result = await apiFetch("/api/sms-logs", {
   567→          method: "POST",
   568→          body: JSON.stringify({
   569→            recipient: savedRecipient,
   570→            message: savedMessage,
   571→          }),
   572→        });
   573→
   574→        // 5. Replace the optimistic entry with the real one
   575→        setSmsLogs((prev) =>
   576→          prev.map((log) =>
   577→            log.id === optimisticId
   578→              ? { ...(result as any), isOptimistic: false }
   579→              : log
   580→          )
   581→        );
   582→
   583→        const finalStatus = (result as any)?.status || "Sent";
   584→        if (finalStatus === "Sent") {
   585→          toast({ title: "✓ SMS Sent", description: `Message delivered to ${savedRecipient}` });
   586→        } else if (finalStatus === "Failed") {
   587→          toast({ title: "SMS Failed", description: `Gateway error for ${savedRecipient}: ${(result as any)?.gatewayResponse || "Unknown error"}`, variant: "destructive" });
   588→        } else {
   589→          toast({ title: "SMS Queued", description: `Message queued for ${savedRecipient} (Pending delivery)` });
   590→        }
   591→      } catch (e: any) {
   592→        // 6. Mark the optimistic entry as Failed
   593→        setSmsLogs((prev) =>
   594→          prev.map((log) =>
   595→            log.id === optimisticId
   596→              ? { ...log, status: "Failed", isOptimistic: false }
   597→              : log
   598→          )
   599→        );
   600→        toast({ title: "Failed", description: e.message || "Failed to send SMS", variant: "destructive" });

   602→        setSmsSending(false);
   603→        // Refresh data from server to ensure consistency
   604→        loadData();
   605→      }
   606→    } else {
   607→      // ─── WhatsApp-like Optimistic UI: Bulk SMS ───
   608→      const rawRecipients = smsBulkRecipients.split(",").map(r => r.trim()).filter(Boolean);
   609→      const uniqueRecipients = [...new Set(rawRecipients)];
   610→      const duplicatesRemoved = rawRecipients.length - uniqueRecipients.length;
   611→
   612→      if (duplicatesRemoved > 0) {
   613→        toast({ title: "Duplicates Removed", description: `${duplicatesRemoved} duplicate number(s) removed from recipient list`, variant: "destructive" });
   614→      }
   615→      if (uniqueRecipients.length === 0) {
   616→        toast({ title: "Error", description: "No valid unique recipients found", variant: "destructive" });
   617→        setSmsSending(false);
   618→        return;
   619→      }
   620→
   621→      // 1. Create optimistic entries for each recipient
   622→      const { charCount: optCharCount, isUnicode: optIsUnicode, segmentCount: optSegCount } = computeClientSmsSegments(smsMessage.trim());
   623→      const optRate = optIsUnicode ? unicodeCostPerSegment : costPerSegment;
   624→      const estimatedCost = optSegCount * optRate;
   625→
   626→      const bulkMessage = smsMessage.trim();
   627→      const bulkCampaignName = campaignName || null;
   628→
   629→      const optimisticEntries = uniqueRecipients.map((recipient, i) => ({
   630→        id: `temp-bulk-${Date.now()}-${i}`,
   631→        recipient,
   632→        message: bulkMessage,
   633→        charCount: optCharCount,
   634→        smsSegmentCount: optSegCount,
   635→        isUnicode: optIsUnicode,
   636→        status: "Sending",
   637→        sentAt: new Date().toISOString(),
   638→        cost: estimatedCost,
   639→        campaignName: bulkCampaignName,
   640→        isOptimistic: true,
   641→      }));

   643→      // 2. Clear form immediately
   644→      setSmsBulkRecipients("");
   645→      setSmsMessage("");
   646→      setCampaignName("");
   647→
   648→      // 3. Add optimistic entries to the top of the logs
   649→      setSmsLogs((prev) => [...optimisticEntries, ...prev]);
   650→
   651→      // 4. Make the API call
   652→      try {
   653→        const result = await apiFetch("/api/sms-logs", {
   654→          method: "POST",
   655→          body: JSON.stringify({
   656→            batchMode: true,
   657→            recipients: uniqueRecipients,
   658→            message: bulkMessage,
   659→            campaignName: bulkCampaignName || undefined,
   660→          }),
   661→        }) as any[];
   662→
   663→        // 5. Replace optimistic entries with real ones
   664→        if (Array.isArray(result)) {
   665→          const resultMap = new Map(result.map((r: any) => [r.recipient, r]));
   666→          setSmsLogs((prev) => {
   667→            // Remove all optimistic entries for this batch
   668→            const withoutOptimistic = prev.filter(
   669→              (log) => !optimisticEntries.some((oe) => oe.id === log.id)

   677→          const pendingCount = result.filter((r: any) => r.status === "Pending").length;

   681→            description: `Sent: ${sentCount}, Failed: ${failedCount}, Pending: ${pendingCount}${duplicatesRemoved > 0 ? ` (${duplicatesRemoved} duplicates removed)` : ''}`,
   682→          });
   683→        }
   684→      } catch (e: any) {

   688→            optimisticEntries.some((oe) => oe.id === log.id)

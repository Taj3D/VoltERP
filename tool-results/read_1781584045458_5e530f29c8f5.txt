    77→const sanitizeInput = (input: string): string => {
    78→  return input.replace(/<[^>]*>/g, '').replace(/\r\n/g, '\n').replace(/  +/g, ' ').trim();
    79→};
    80→
    81→const sanitizeCSVInput = (input: string): string => {
    82→  return input
    83→    .replace(/<script[^>]*>[\s\S]*?<\/script>/gi, '')
    84→    .replace(/javascript:/gi, '')
    85→    .replace(/on(error|load|click|mouseover|focus|blur)=/gi, '')
    86→    .replace(/<iframe[^>]*>[\s\S]*?<\/iframe>/gi, '')
    87→    .replace(/eval\s*\(/gi, '')
    88→    .replace(/document\.cookie/gi, '')
    89→    .replace(/window\.location/gi, '')
    90→    .trim();
    91→};
    92→
    93→// Field length validation
    94→const validateFieldLength = (value: string, fieldType: 'name' | 'numeric' | 'percentage'): string | null => {
    95→  if (fieldType === 'name' && value.length > 100) return 'Name fields must be max 100 characters';
    96→  if (fieldType === 'numeric' && value.replace(/[.,\-]/g, '').length > 15) return 'Numeric fields must be max 15 digits';
    97→  if (fieldType === 'percentage' && value.length > 5) return 'Percentage fields must be max 5 characters';
    98→  return null;
    99→};

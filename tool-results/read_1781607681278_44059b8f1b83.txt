    43→// Safe formatters — centralized utility guarantees Latin digits
    44→import { fmtCurrency as _fmtCurrencyVal, fmtNumber as _fmtNumberVal, fmtBDT as _fmtBDT } from "@/lib/number-format";
    45→
    46→const fmt = (v: any, type?: string) => {
    47→  if (v === null || v === undefined || v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v || "—";
    48→  if (type === "currency") return _fmtBDT(Number(v));
    49→  if (type === "date") { if (!v) return "—"; const dt = new Date(v); return isNaN(dt.getTime()) ? "—" : dt.toLocaleDateString("en-GB", { day: "2-digit", month: "short", year: "numeric" }); }
    50→  if (type === "boolean") return v ? "Active" : "Inactive";
    51→  if (type === "number") return _fmtNumberVal(Number(v));
    52→  return String(v);
    53→};
    54→
    55→const fmtCurrency = (v: any) => {
    56→  if (v === null || v === undefined) return "—";
    57→  if (v === "N/A (Audit Mode)" || v === "N/A (Restricted)") return v;
    58→  return _fmtBDT(Number(v));
    59→};

   775→  const computeOrderTotals = (lines: any[], discountPercent: number, vatPercentage: number) => {
   776→    const subTotal = lines.reduce((s: number, l: any) => {
   777→      const qty = Number(l.quantity) || 0;
   778→      const rate = Number(l.rate) || 0;
   779→      const lineDisc = Number(l.discountPercent) || 0;
   780→      const lineTotal = qty * rate * (1 - lineDisc / 100);
   781→      return s + lineTotal;
   782→    }, 0);
   783→    const discountAmt = subTotal * (discountPercent / 100);
   784→    const afterDiscount = subTotal - discountAmt;
   785→    const vatAmt = afterDiscount * (vatPercentage / 100);
   786→    const grandTotal = afterDiscount + vatAmt;
   787→    return { subTotal: Math.round(subTotal * 100) / 100, discountAmt: Math.round(discountAmt * 100) / 100, vatAmt: Math.round(vatAmt * 100) / 100, grandTotal: Math.round(grandTotal * 100) / 100 };
   788→  };
